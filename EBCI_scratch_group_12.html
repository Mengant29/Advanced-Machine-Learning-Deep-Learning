def cv_per_subj_test(x, y, params, path_to_subj, test_on_last_block=False, plot_fold_history=False, write=False):
    if write:
        model_path = os.path.join(path_to_subj, 'checkpoints')
    else:
        model_path = None
    best_val_epochs = []
    best_val_aucs = []
    folds = 4  # To preserve split as 0.6 0.2 0.2
    if test_on_last_block:
        x_tr, y_tr, x_tst, y_tst = separte_last_block(x, y, test_size=0.2)
    else:
        x_tr, y_tr = x, y

    cv = StratifiedKFold(n_splits=folds, shuffle=True, random_state=0)
    cv_splits = list(cv.split(x_tr, y_tr))
    for fold, (train_idx, val_idx) in enumerate(cv_splits):
        if write:
            fold_model_path = os.path.join(model_path, '%d' % fold)
            os.makedirs(fold_model_path)
            model_path_ = os.path.join(fold_model_path, 'model{}'.format(fold))
        else:
            model_path_ = None
        x_tr_fold, y_tr_fold = x_tr[train_idx], y_tr[train_idx]
        x_val_fold, y_val_fold = x_tr[val_idx], y_tr[val_idx]

        val_history, fold_model = train_model_eegnet(x_tr_fold, y_tr_fold, params, (x_val_fold, y_val_fold), epochs=200,
                                                     batch_size=32, shuffle=True,
                                                     model_path=model_path_)
        best_val_epochs.append(np.argmax(val_history['val_auc']) + 1)  # epochs count from 1 (not from 0)
        best_val_aucs.append(np.max(val_history['val_auc']))
        if plot_fold_history:
            single_auc_loging(val_history, 'fold %d' % fold, fold_model_path)

    if test_on_last_block:
        test_history, final_model = train_model_eegnet(x_tr, y_tr, params, epochs=int(np.mean(best_val_epochs)),
                                                       validation_data=(x_tst, y_tst), batch_size=32, shuffle=True,
                                                       model_path=model_path)
        single_auc_loging(test_history, 'test_history', path_to_save=path_to_subj)
        
    if write:
        with codecs.open('%s/res.txt' % path_to_subj, 'w', encoding='utf8') as f:
            f.write(u'Val auc %.02f%.02f\n' % (np.mean(best_val_aucs), np.std(best_val_aucs)))
            if test_on_last_block:
                f.write('Test auc naive %.02f\n' % (test_history['val_auc'][-1]))
                return test_history['val_auc'][-1], final_model
    
    return np.mean(best_val_aucs), np.std(best_val_aucs)

def train_per_subj_test(x, y, params, path_to_subj, test_on_last_block=False):
    
    if test_on_last_block:
        x_tr, y_tr, x_tst, y_tst = separte_last_block(x, y, test_size=0.2)
        validation = (x_tst, y_tst)
    else:
        x_tr, y_tr = x, y
        validation = None
    
    
    test_history, final_model = train_model_eegnet(x_tr, y_tr, params, epochs=200,
                                                       validation_data=None, batch_size=32, shuffle=True,
                                                       model_path=os.path.join(path_to_subj, "model"))

        single_auc_loging(test_history, 'test_history', path_to_save=path_to_subj)

def train_cv(subjects, subjs_test_stats, test_on_last_block, params, write=False): #subjects est un dataset charg comme prcdement. 
    subjs_test_stats = {}
    for train_subject in subjects.keys():
        if write:
            path_to_subj = prepare_dirs(experiment_res_dir, train_subject)
        else:
            path_to_subj = None
        x = subjects[train_subject][0]
        x = x.transpose(0, 2, 1)[:, np.newaxis, :, :]
        y = subjects[train_subject][1]
        test_auc, std_test_auc = cv_per_subj_test(x, y, params, path_to_subj, test_on_last_block=test_on_last_block, write=write)
        subjs_test_stats[train_subject] = (test_auc, std_test_auc)
    return subjs_test_stats

def objective(trial):

    resample_val = trial.suggest_int('resample_to', 380, 420)
    custom_params = {}
    custom_params['resample_to'] = resample_val
    
    custom_params['D'] = 3
    custom_params['F1'] = 12
    custom_params['dropoutRate1'] = 0.52
    custom_params['dropoutRate2'] = 0.36
    custom_params['lr'] = 0.00066
    custom_params['norm_rate'] = 0.275
    
#     custom_params['D'] = trial.suggest_int('resample_to', 2, 4)
#     custom_params['F1'] = trial.suggest_int('resample_to', 5, 15)
    custom_params['dropoutRate1'] = trial.suggest_uniform('dropoutRate1', 0.2, 0.8)
    custom_params['dropoutRate2'] = trial.suggest_uniform('dropoutRate2', 0.2, 0.8)
    custom_params['lr'] = trial.suggest_loguniform('lr', 1e-5, 1e-4)
    
    subjects = createSubjectsWithResample(resample_val)
    subjs_test_stats = {}
    mean_AUCs = train_cv(subjects, subjs_test_stats, test_on_last_block=False, params=custom_params)    
    
    return mean_AUCs

def model_cv(subj, x, y, cv_num):
    model_dict = torch.load(os.path.join("./res", str(subj), "checkpoints", str(cv_num), "model{}".format(cv_num)))
    kernLength1 = best_params.get('kernLength1',64)
    kernLength2 = best_params.get('kernLength2',16)
    F2 = best_params.get('F2',best_params['F1']*best_params['D'])

    model = EEGNet_experimental(nb_classes=2, Chans=x.shape[2], Samples=x.shape[3],
                                   dropoutRates=(best_params['dropoutRate1'], best_params['dropoutRate2']),
                                   kernLength1=kernLength1, kernLength2=kernLength2, poolKern1=4,
                                   poolKern2=8,
                                   F1=best_params['F1'],
                                   D=best_params['D'], F2=F2)

    # using gpu if possible (as we are not going to train anymore, but just predict, not very important)
    device = torch.device('cuda:0' if torch.cuda.is_available() else 'cpu')
    model.to(device)
    
    skf = StratifiedKFold(n_splits=4, shuffle=True, random_state=0)
    train_idx, val_idx = list(skf.split(x, y))[cv_num]
    
    x_tensor = torch.from_numpy(x[train_idx]).float().to(device)
    x_val_tensor = torch.from_numpy(x[val_idx]).float().to(device)
    model.load_state_dict(model_dict)
    
    with torch.no_grad():
        model.eval()
        features, _ = model.forward_features(x_tensor)
        features_val, _ = model.forward_features(x_val_tensor)
        y_full = torch.nn.Softmax(dim=1)(torch.sigmoid(model.forward(x_val_tensor)))[:, 1]
    return model.flatten(features).numpy(), model.flatten(features_val).numpy(), y_full.numpy()

def compare_cv(clf, x, y, n_split=4):
    
    auc_clf = []
    auc_full = []
    for cv in range(n_split):
        features, features_val, y_full_forward = model_cv(25, x, y, cv)
                
        skf = StratifiedKFold(n_splits=n_split, shuffle=True, random_state=0)
        train_idx, val_idx = list(skf.split(x, y))[cv]
        
        clf.fit(features, y[train_idx])
        
        auc_clf.append(roc_auc_score(y[val_idx], clf.predict(features_val)))
        auc_full.append(roc_auc_score(y[val_idx], y_full_forward))
    auc_clf = np.array(auc_clf)
    auc_full = np.array(auc_full)
    
    return (auc_clf.mean(), auc_clf.std()), (auc_full.mean(), auc_full.std())
